```python
def load_closed_yaml() -> list[dict[str, Any]]:
    """Load closed.yaml. Returns list of closed tasks.

    Handles the {meta, tasks} format.
    Falls back to legacy done.yaml if closed.yaml doesn't exist.
    Returns [] if neither file exists.
    """
    path = _resolve_tasks_dir() / CLOSED_FILE_NAME
    if not path.exists():
        old_path = _resolve_tasks_dir() / "done.yaml"
        if old_path.exists():
            raw = yaml.safe_load(old_path.read_text(encoding="utf-8")) or []
            if isinstance(raw, dict):
                return raw.get("tasks", [])
            return raw if isinstance(raw, list) else []
        return []
    raw = yaml.safe_load(path.read_text(encoding="utf-8")) or []
    if isinstance(raw, dict):
        return raw.get("tasks", [])
    return raw if isinstance(raw, list) else []


def save_closed_yaml(closed_list: list[dict[str, Any]]) -> None:
    """Write closed.yaml with meta wrapper and header.

    Oldest-first (chronological order). Meta tracks total_closed counter.
    """
    path = _resolve_tasks_dir() / CLOSED_FILE_NAME
    path.parent.mkdir(parents=True, exist_ok=True)
    meta = {"total_closed": 0}
    if path.exists():
        existing_raw = yaml.safe_load(path.read_text(encoding="utf-8")) or {}
        if isinstance(existing_raw, dict):
            meta = existing_raw.get("meta", meta)
    meta["total_closed"] = len(closed_list)
    data = {"meta": meta, "tasks": _strip_none_fields_from_list(closed_list)}
    content = CLOSED_HEADER + yaml.safe_dump(data, default_flow_style=False, sort_keys=False,
                                              allow_unicode=True)
    path.write_text(content, encoding="utf-8")
```

✅ Success: `load_closed_yaml()` and `save_closed_yaml()` exist. Fallback reads old `done.yaml`. Old `load_done_yaml` / `save_done_yaml` are gone.

```python
def close_task(task_id: str, note: str | None = None) -> dict[str, Any]:
    """Move a task from tasks.yaml to closed.yaml. Preserves its status.

    Sets closed=True, adds 'closed_at' timestamp.
    Appends optional --note to the existing notes field.
    Returns the closed task dict.
    Raises ValueError if task not found or already closed.
    """
    meta, tasks = load_tasks_yaml()
    task = _find_task_by_id(tasks, task_id)
    if task is None:
        # Check if task is already closed
        closed_list = load_closed_yaml()
        closed_task = _find_task_by_id(closed_list, task_id)
        if closed_task is not None:
            raise ValueError(f"Task already closed: {task_id}")
        raise ValueError(f"Task not found: {task_id}")

    # Remove from active list
    tasks.remove(task)

    # Update the task — PRESERVE status, set closed flag
    task["closed"] = True
    task["closed_at"] = _now_iso()
    task["updated"] = _now_iso()
    if note:
        existing_notes = task.get("notes", "")
        task["notes"] = f"{existing_notes}\n{note}".strip() if existing_notes else note

    # Save active list
    save_tasks_yaml(meta, tasks)

    # Append to closed.yaml (oldest-first: append to bottom)
    closed_list = load_closed_yaml()
    closed_list.append(task)
    save_closed_yaml(closed_list)

    return task
```

✅ Success: `close_task()` sets `closed=True`, preserves status. Uses `load_closed_yaml`/`save_closed_yaml`. No `complete_task` remains.

> it moves to `closed.yaml` without modifying the status.
